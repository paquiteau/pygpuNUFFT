152inline void freeTotalDeviceMemory(void *ptr, ...)
153{
154 va_list list;
155 va_start(list, ptr);
156 void *p = ptr;
157 int i = 0;
158 while (true)
159 {
160 if (p == NULL)
161 break;
162 // printf("free dev ptr...%p %d\n",p,i);
163 freeDeviceMem((void *)p);
164 i++;
165 p = va_arg(list, void *);
166 }
167 cudaDeviceSynchronize();
168 if (DEBUG)
169 printf("%d device pointers freed\n", i);
170
171 va_end(list);
172}
