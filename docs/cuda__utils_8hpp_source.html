166inline void freeTotalDeviceMemory(void *ptr, ...)
167{
168 va_list list;
169 va_start(list, ptr);
170 void *p = ptr;
171 int i = 0;
172 while (true)
173 {
174 if (p == NULL)
175 break;
176 // printf("free dev ptr...%p %d\n",p,i);
177 freeDeviceMem((void *)p);
178 i++;
179 p = va_arg(list, void *);
180 }
181 cudaDeviceSynchronize();
182 if (DEBUG)
183 printf("%d device pointers freed\n", i);
184
185 va_end(list);
