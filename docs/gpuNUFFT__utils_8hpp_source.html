286__inline__ __device__ __host__ DType
287 calculateDeapodizationValue(int coord, DType grid_width_inv,
288 int kernel_width, DType beta)
289{
290 DType poly =
291 sqr((DType)M_PI) * sqr(kernel_width) * sqr(grid_width_inv) * sqr(coord) -
292 sqr(beta);
293 DType val;
294 // sqrt for negative values not defined -> workaround with sinh
295 if (poly >= 0)
296 val = sin(sqrt(poly)) / sqrt(poly);
297 else
298 val = sinh(sqrt((DType)-1.0 * poly)) / sqrt((DType)-1.0 * poly);
299
300 return val;
301}

309__inline__ __device__ __host__ DType
310 calculateDeapodizationAt(int x, int y, int z, IndType3 width_offset,
311 DType3 grid_width_inv, int kernel_width,
312 DType beta, DType norm_val)
313{
314 int x_shifted = x - (int)width_offset.x;
315 int y_shifted = y - (int)width_offset.y;
316 int z_shifted = z - (int)width_offset.z;
317
318 DType val_x = calculateDeapodizationValue(x_shifted, grid_width_inv.x,
319 kernel_width, beta);
320 DType val_y = calculateDeapodizationValue(y_shifted, grid_width_inv.y,
321 kernel_width, beta);
322 DType val_z = calculateDeapodizationValue(z_shifted, grid_width_inv.z,
323 kernel_width, beta);
324
325 return val_x * val_y * val_z / norm_val;
326}
